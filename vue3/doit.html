# props--父传子

<!-- parent.vue -->
<template>
  <div class="parent">
    <h1>这是父组件</h1>
    <Child name="test"></Child>
  </div>
</template>

<script lang="ts" setup>
import Child from './Child.vue'    //在setup中无需注册组件

</script>

<!-- Child.vue -->
<template>
  <div>
    <h2>这是子组件:{{name}}</h2>
  </div>
</template>

<script lang="ts" setup> 
//需要使用defineProps方法接收父组件传递过来的数据
//defineProps是vue3提供的，直接使用不需要引入
let props = defineProps(['name'])    
//打印props可以看出defineProps返回的是一个代理对象Proxy,在模板中使用{{name}}来使用数据。props.name也可以
//在script中还是需要通过props.name来使用

/**
    defineProps({
        name:{
            type:String,
            required:true
        }
    })
    
    ts写法：非必填用？即可 name?:string
    difineProps<{
    	name:string
    }>()
    
    
    默认值写法：配置wiehDefaults
    type Props = {
    	msg?: string
  		labels?: string[]
    }
    
    const props = withDefaults(defineProps<Props>(), {
      msg: 'hello',
      labels: () => ['one', 'two']
    })
    
*/


//父组件直接调用子组件方法也可以给子组件传值
//但要注意实例有可能获取不到子组件的属性或方法，因为子组件使用了setup模板，且没有用defineExpose()将其暴露

</script>

<!--兼容vue2的写法-->
<script>
export default {
  props: ['name']
}
</script>

# 自定义事件--子传父

<!-- parent.vue -->
<template>
  <div class="parent">
    <h1>这是父组件</h1>
    <!-- 
      在vue2中，组件需要.native把事件变为原生dom事件
      vue3中，原生dom事件无论放在标签，或是组件标签上都是原生dom事件，实际是给组件的根节点绑定
    -->
    <Child @test="handlerTest"></Child>   
      <!--假设子组件触发的方法名为updateTest，那父组件绑定时可以@updateTest,也可以@update-test,@update-Test -->
  </div>
</template>

<script setup>
import Child from './Child.vue'

const handlerTest = (param, param2) => {
  console.log('xx', param, param2);
}
</script>

<!-- Child.vue -->
<template>
  <div>
    <h2>这是子组件</h2>
    <button @click="sendMes">传数据回去</button>
    <button @click="$emits('test','ak','aspea')">传数据回去</button>
  </div>
</template>

<script setup>
//需要使用defineEmits返回一个函数用于触发自定义事件
//defineEmits是vue3提供的，直接使用不需要引入
let $emits = defineEmits(['test'])   //数组里好像不传只是会在控制台弹出警告.如果写了原生事件，那该事件就会当作自定义事件
//Component emitted event "test" but it is neither declared in the emits option nor as an "onTest" prop.

/**
   ts写法：   (e:事件名,参数名:类型,参数名:类型): void
   const $emit = defineEmits<{
   		(e:'test',name:string,age:number): void
   }>

*/

const sendMes = () => {
  $emits('test','name','age')
}
</script>

# 全局事件总线

通过mitt插件实现，类似vue2中的pubsub，也是订阅发布模式

npm i -s mitt

在src目录下创建目录bus，创建文件index.ts

import mitt from 'mitt'
const $bus = mitt();  //方法执行返回bus对象
export default $bus;

<!-- Child.vue 子组件1-->
<template>
  <div>
    <h1>这是子组件1</h1>
    <hr />
    <button @click="sendMes">给子组件2传递数据</button>
  </div>
</template>

<script lang="ts" setup>
import $bus from '../bus'

const sendMes = ()=>{
    $bus.emit('mes','alcohol','free')
}
</script>

<!-- Child2.vue 子组件2-->
<template>
  <div>
    <h1>这是子组件2</h1>
  </div>
</template>

<script lang="ts" setup>
import $bus from "../bus";
import { onMounted } from "vue";
//组件挂载完毕，绑定一个事件接收将来兄弟组件传来的数据
onMounted(() => {
  //第一个参数为事件类型，第二个为回调函数
  $bus.on("mes", (data) => {
    console.log("收到数据:", data);
  });
});
</script>

# v-model --父子数据同步

<!-- parent.vue -->
<template>
  <div>
    <input type="text" v-model="money" />
    <Child v-model="money"></Child>
      <!-- v-model在组件上使用相当于
        给子组件传递props[modelValue]=绑定值
        给子组件绑定自定义事件@update:modelValue

		==在vue2中是通过.sync语法糖实现父子数据同步
		//父元素
		<Child :num.sync="num"></Child>
		//等同于 给子元素绑定num值，并绑定自定义事件update:变量名
		<Child :num="num" @update:num="xxx"></Child>

		==vue2里v-model也可以实现，但与vue3中可同时绑定多个v-model
		https://v2.cn.vuejs.org/v2/guide/components-custom-events.html#%E8%87%AA%E5%AE%9A%E4%B9%89%E7%BB%84%E4%BB%B6%E7%9A%84-v-model
     -->
      <!-- 绑定多个值 -->
      <Child2 v-model:pageNo="pageNo" v-model:pageSize='pageSize'></Child2>
  </div>
</template>

<script setup lang="ts">
import Child from "./components/Child.vue";
import Child2 from "./components/Child2.vue";
import { ref } from "vue";
let money = ref(666);
let pageNo = ref(1);
let pageSize = ref(10);
</script>

<!-- Child.vue 子组件1-->
<template>
  <div>
    <h1>这是子组件1,money:{{ modelValue }}</h1>
    <hr />
    <button @click="add">增加</button>
  </div>
</template>

<script lang="ts" setup>
const props = defineProps(["modelValue"]);
const $emit = defineEmits(["update:modelValue"]);
const add = () => {
  $emit("update:modelValue", props.modelValue + 100);
};
</script>

<!-- Child2.vue 子组件2-->
<template>
  <div>
    <h1>这是子组件2</h1>
    <h2 @click="$emit('update:pageNo',pageNo+1)">pageNo:{{pageNo}}</h2>
    <h2 @click="$emit('update:pageSize',pageSize+1)">pageSize:{{pageSize}}</h2>
  </div>
</template>

<script lang="ts" setup>
let props= defineProps(['pageNo','pageSize']);
let $emit = defineEmits(['update:pageNo','update:pageSize'])
</script>

# useAttrs --获取组件标签属性与事件

<!-- parent.vue -->
<template>
  <div>
      <!-- :="{a:1}" 就会给标签绑定a=1 -->
      <Child2 :pageNo="pageNo" :pageSize='pageSize' :="{a:1,b:2}"></Child2>
  </div>
</template>

<!-- Child2.vue 子组件2-->
<script lang="ts" setup>
import { useAttrs } from "vue";
let $attrs = useAttrs()  //可获取到父组件传递的标签属性    
    
//但是如果用defineProps接收的话，$attrs里的相应的数据就获取不到
let props= defineProps(['pageNo','pageSize']);

</script>

# ref与$parent

ref：获取真实的dom节点，获取子组件实例(通常在标签上绑定非字符串数值时需要加上冒号，而绑定ref时却不需要加冒号

官网的说法是引用将存储在与名字匹配的ref里，非常反直觉，如果加冒号是函数式的绑定方法，函数第一个参数是该元素ref值，在函数中赋值给想要存储的变量中)

$parent：在子组件中获取父组件的实例

<!-- parent.vue -->
<template>
  <div>
    <Child ref="childRef"></Child>
  </div>
</template>

<script setup lang="ts">
import Child from "./components/Child.vue";
import { ref, onMounted } from "vue";
let childRef = ref(null);
let age = ref(100)
onMounted(()=>{
  console.log("xxx", childRef.value.money);
})
 
    defineExpose({age})
</script>

<!-- Child.vue -->
<template>
  <div>
    <h1>这是子组件1,money:{{ money }}</h1>
    <button @click="handler($parent)">test</button>
  </div>
</template>

<script lang="ts" setup>
import { ref } from "vue";
let money = ref(666);

//在<script setup>组件或$parent获取的实例不会暴露绑定的数据,需要defineExpose对外暴露.
defineExpose({
  money,
});
    
const handler = ($parent) => {   //$parent目前好像只能在事件回调中通过参数获取
	console.log("??", $parent.age);
};
</script>

# provide与inject--隔代组件数据传递

<!-- parent.vue -->
<template>
  <div class="parent">
    <Child></Child>
  </div>
</template>

<script lang="ts" setup>
import { ref,provide } from 'vue'
import Child from './Child.vue'  
let car = '帕拉梅拉'

provide('test1',car)   //第一个参数是数据的key，第二个是值
</script>

<!-- 所有后代组件 -->
<template>
  <div>
    <h2>这是后代组件:{{car}}</h2>
  </div>
</template>

<script lang="ts" setup> 
import { inject } from 'vue'
    
let car = inject('test1') //注入祖先提供的数据,获取的值可以修改，就和自己创建的ref一样.可填入第二个参数，意味注入的数据为空时，默认值为填入的第二个参数
</script>

# pinia

<!-- 
	vuex:集中式管理状态容器，核心概念：store、mutations、actions、getters、modules
	pinia：核心概念：store、actions、getters
-->

选择式api：

/** store/main.ts */
import { createPinia } from 'pinia'
let pinia = createPinia();
export default pinia;

/** 入口文件main.ts */
import pinia from '@/store/index'
createApp(App).use(router).use(pinia).mount('#app')

/** store/modules/info.ts */
import { defineStore } from 'pinia'

//选择式api
let useInfoStore = defineStore('Info', {
  state: () => {
    return {
      age:1
    }
  },
  //异步|逻辑
  actions: {
    //登录
    userLogin(a,b){
      console.log('info',a,b,this.token);
    },
    add(){  //可携带多个参数。没有context上下文，通过this调用自身属性
      this.age++
    }
  },
  getters: {
	total(){
        return this.age+1
    }
  }
})

export default useInfoStore;

<!--任意组件-->
<script setup lang='ts'>
import useInfoStore from '@/store/modules/info'
let infoStore = useInfoStore();    //通过infoStore.xxx调用仓库里的属性和方法
    //修改状态-- 可以直接infoStore.age++ 或调用 infoStore.$patch({ age:2 }) 或者下面这种函数式，但和react更新不同，没有return
    // userStore.$patch((state) => {
  	//   state.token=6
    // })
    
    //通过调用store里action定义的方法修改
    //调用$reset会恢复state的默认值
</script>

组合式api：

//组合式api
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'   //ref相当于选择式里的state、computed相当于getters
let useGoodsStore = defineStore('goods', () => {
  let goods = ref(100)
  let total = computed(() => {
    return goods.value + 100
  })
  const update = () => {
    goods.value++
  }
  //必须返回一个对象：属性和方法可以提供给组件使用
  return {
    goods,
    total,
    update
  }
})

export default useGoodsStore;

<!--任意组件-->
<script setup lang='ts'>
import useGoodsStore from '@/store/goods'

# nextTick补充

import { nextTick } from 'vue';

nextTick(() => {
  // 在 DOM 更新循环结束后执行的代码
  // 这里可以操作已经更新的 DOM
});

nextTick().then(()=>{ 
    //xxx 
})
//.then意味着可以使用await
await nextTick();
//xxx

# 按需导入element plus/icon

需要安装unplugin-icons 、@iconify/json、 unplugin-auto-import，第三个前面已经安装过，官网里是没有提到第二个的

pnpm i unplugin-icons @iconify/json -D

修改vite.config.ts：

import IconsResolver from "unplugin-icons/resolver";
import Icons from 'unplugin-icons/vite'

export default defineConfig(({ command, mode }) => {
  return {
    plugins: [
      vue(),
      AutoImport({
        resolvers: [
          ElementPlusResolver(),
          IconsResolver({   //自动导入图标
            prefix: "Icon",
          }),
        ],
      }),
      Components({
        resolvers: [
          ElementPlusResolver(),
          IconsResolver({  //自动注册图标
            enabledCollections: ["ep"],
          }),
        ],
      }),
      Icons({        //自动安装
        autoInstall: true,
      }),
    ],
  };
});

//这种图标引入方式是无效的
<el-icon>
    <Plus />
</el-icon>

//要改为下面这种,正确引入的时候，可以看到它会自动下载一个依赖@iconify-json/ep
<el-icon>
    <i-ep-plus />
</el-icon>

# axios封装

pnpm install axios

目的：

使用请求拦截器，可以在请求中处理一些业务（请求头添加公共参数、开始进度条）

使用响应拦截器，可以在响应后处理一些业务（关闭进度条、简化服务器返回的数据、处理http网络错误）

创建src/utils/request.ts

import axios from 'axios'
import { ElMessage } from 'element-plus'

let request = axios.create({
    baseURL: '/api',//请求的基础路径会携带/api
    timeout: 5000,//超时时间设置
});

//添加请求拦截器
request.interceptors.request.use((config) => {

    //返回配置的对象
    return config
})



//响应拦截器
request.interceptors.response.use((response) => {
    return response;
}, (error) => {
    let message = ''
    let status = error.response.status;
    switch (status) {
        case 400: message = 'token已过期'; break;
        case 403: message = '无权访问'; break;
        case 404: message = '请求地址错误'; break;
        case 401: message = '服务器出错'; break;
        default: message = '网络出错'; break;
    }
    ElMessage({
        type: 'error',
        message
    })
    return Promise.reject(error)  //return Promise.reject .catch或try/catch才能捕捉错误
})

export default request;

# element plus表单的resetFields不生效

放在dialog里的form，以编辑模式打开dialog后关闭出发close的resetFields方法，再以新建模式打开dialog发现数据没有清除。

如果是以新建=》编辑=》新建的方式，发现其实是生效的。

原因是在编辑=》新建 方式中打开编辑模式的时候，直接对form表单的数据进行了赋值，form表单就把赋值展示的数据以为是初始值，所以重置的就是那些数据。

应该在打开dialog时用nextTick进行赋值

<el-form  @submit.native.prevent>
</el-form>

# element plus 组件打开dialog或drawer调用focus无法自动获取里面input的焦点

按常理，setTimeout和nextTick都能获取到inputRef实例，确实也能打印出来，但是nextTick里的聚焦却不起作用，需要多套一层nextTick。

解决办法：

1、focus方法在两层nextTick中调用

2、focus方法在setTimeout中调用

3、focus方法在drwer或dialog组件中方法opened中调用

Dialog 、Drawer的内容是懒渲染的——在被第一次打开之前，传入的默认 slot 不会被立即渲染到 DOM 上。 因此，如果需要执行 DOM 操作，或通过 ref 获取相应组件，请在 open 事件回调中进行。
