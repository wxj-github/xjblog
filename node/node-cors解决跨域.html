npm install cors

全局开启：index.js或者app.js中

const express = require('express')
const path = require('path')
const cors = require('cors')   // ++++++

const app = express()
app.use(cors())           //  ++++++++
var test = require('./test')
app.use('/test', test)
console.log('8010接口启动!!')
app.listen(8010)

个别接口开启，增加白名单

var express = require('express');
var router = express.Router();
const cors = require('cors')

var allowlist = ['http://localhost:3000', 'http://example2.com']   //白名单

var corsOptionsDelegate = function (req, callback) {
    var corsOptions;
    //console.log('req.header', req.header('Origin'));
    if (allowlist.indexOf(req.header('Origin')) !== -1) {
        corsOptions = { origin: true } // 同意跨域
    } else {
        corsOptions = { origin: false } // 拒绝跨域
    }
    callback(null, corsOptions) // callback expects two parameters: error and options
}


router.get('/all', cors(corsOptionsDelegate), async function (req, res, next) {
    res.json({
        'flag': true
    })

});

module.exports = router;
