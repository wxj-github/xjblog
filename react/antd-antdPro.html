# Form

属性：

# form:

经Form.useForm()创建的 form 控制实例,类组件只能ref控制实例了

//通过Form.useForm()创建表单容器,比起useRef可以少打个current
const App=()=>{
    const [form] = Form.useForm();  //只能用于函数组件
    // form.resetFields()   form.setFieldsValue({})  form.validateFields()
    return (<Form  form={form}></Form>)
}

# onFinish:

提交表单且数据验证成功后回调事件,第一个参数为表单值。之前都是用ref.validateFields触发验证的(校验不通过抛出异常，配合try catch)

import { Form, Button } from 'antd';

const MyForm = () => {
  const [form] = Form.useForm();

  const onFinish = (values) => {
    console.log('Form values:', values);
  };

  return (
    <Form form={form} onFinish={onFinish}>
      <Form.Item name="name" label="Name">
        <input />
      </Form.Item>
      <Form.Item name="email" label="Email">
        <input />
      </Form.Item>
      //通过htmlType='submit'可设定此按钮为提交表单。不需要放在Form.Item里，但必须放在Form里
      <Button type="primary" htmlType="submit">  
          Submit
        </Button>
    </Form>
    <Button onClick={() => form.submit()}>Submit</Button>  //也可以通过form.submit触发提交，就不必放在form组件里
  );
};

# Form.Item

属性：

# valuePropName：

子节点受控值的属性，默认为value，以此属性作为受控属性,Form表单会读取此值作为该item的值。

设定的值会当作属性传入子节点的props中。如果没有使用trigger属性，就通过props.onChange(newValue)来更新这个valuePropName的值

{/** 平时的一些表单都是通过value属性获取其值，但Switch是checked、Upload是fileList，table是dataSource，需要正确指定值的获取属性，所以需要valuePropName */}
<Form>
      <Form.Item label="InputNumber" name="input-number">
          <Input />   
       </Form.Item>
	  <Form.Item name="switch" label="Switch" valuePropName="checked">
        <Switch />
      </Form.Item>
      <Form.Item
        name="upload"
        label="Upload"
        valuePropName="fileList"
      >
        <Upload name="logo" action="/upload.do" listType="picture">
        </Upload>
      </Form.Item>
    <Form.Item label="two" name="two" valuePropName="xixi">  
        {/**按理来说antd里Input应该是封装调用了Props.onChange来更新valuePropName传入的值，但是并不能正确获取该输入框的值，所以非自定义组件或官方文档出现过的，不建议使用这个属性*/}
          <Input />
    </Form.Item>
</Form>

# trigger:

设置收集字段值变更的时机,默认为onChange。设定的值会变成一个同名函数传入子节点的props中，在自定义组件更新表单时触发该同名函数即可更新表单Item的值

import { Form, Input, Button } from "antd";
import FormTest from "./FormTest";

export default function One() {

  const onfinish = (val) => {
    console.log("finish", val);
    console.table(val);
  };

  return (
    <div>
      <Form onFinish={onfinish}>
        <Form.Item
          label="name"
          name="name"
          rules={[{ required: true, message: "required" }]}
        >
          <Input />
        </Form.Item>
        <Form.Item
          label="test"
          name="test"
          valuePropName="testValue2"   //这个不加也不影响正确获取表单test的值
          trigger="setTestValue2"	//这里也可以不修改，Form.Item的子节点将需要的更新的值通过props.onChange(newValue)更新也可以
          rules={[
            {
              required: true,
              message: 'required'
            },
          ]}
        >
          <FormTest />
        </Form.Item>
        <Form.Item label="two" name="two" valuePropName="xixi">
          <Input />
        </Form.Item>
        <Button htmlType="submit">submit</Button>
      </Form>
    </div>
  );
}

# request

用于请求数据

import {getData} from '@/server.js'
function App(){
    
    const [pagination,setPagination] = useState({
        current:1,
        pageSize:5
    })
    
    const handlePageChange = (current,pageSize)=>{
        setPagination({current,pageSize})
    }
    
	const columns = [
        {
            title:'序号',
            align:'center',
            width:50,
            render:(text,record,index)=>{
                return (pagination.current-1)*pagination.pageSize + index + 1
            }
        }
    ];
    
    return (<ProTable 
       columns={columns}
       rowKey="id"
       pagination={{ 
        	pageSizeOptions:[5,10,15,20],
            ...pagination,
            onChange:handlePageChange
        }}     //页码页数不传默认为1，20
       request={async(params,sort,filter)=>{  
        //params值为current和pageSize,如果搜索表单有输入，相应字段也会在params里
		const res = await getData(params);
         return {  //返回数据前自带loading
             data:res.data.data,//展示的数据
             success:true,  //必须返回true，不然 table 会停止解析数据，即使有数据
             total:res.data.total  //不传会使用 data 的长度，如果是分页一定要传
         }
    }}></ProTable>)

}

# actionRef

Table action 的引用，便于触发Table事件.如

  • reload 刷新
  • reloadAndRest 刷新并清空，页码也会重置，不包括表单
  • reset 重置到默认值，包括表单
  • clearSelected 清空选中项
  • startEditable（rowkey） 开始编辑
  • cancelEditable（rowkey）结束编辑
import { useRef } from 'react';
function App(){
    const actionRef = useRef();
 	// actionRef.current?.reload() 刷新
    return <ProTable actionRef={actionRef}></ProTable>
}
