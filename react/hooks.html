# hooks的setState实现获取最新值

光看这标题好像有点抽象看不懂

/*
	在类组件中，是通过this.setState({},()=>{})来更新状态值,里面可以传入第二个参数作为回调，回调是可以拿到最新的状态的
	我就希望useState也有这种效果，而不是通过useEffect里去执行最新的回调
*/

import { useState } from "react";

const useStateWithCallback = (initalValue) => {
  const [state, setState] = useState(initalValue);

  const set = (value, callback) => {
    //console.log("set", value, callback);
    if (typeof value === "function") {   //这里是考虑到了setState时的函数时写法
      setState((prevState) => {
        const result = value(prevState);
        //console.log("pre and result", prevState, result);
        callback && callback(result);
        return result;
      });
    } else {
      callback && callback(value);
      setState(value);
    }
  };

  return [state, set];
};

export default useStateWithCallback;

使用方法：

import { useState, useEffect } from "react";
import useStateWithCallback from "./useStateWithCallback";

export default function Count(props) {
  const [sum, setSum] = useStateWithCallback(0);

  return (
    <div>
      <h2>sum is {sum}</h2>
      <button onClick={() => setSum(sum + 2,(val)=>{  //需要用形参接收最新值，而不是直接用sum
        console.log('获取最新值拉',val);
      })}>+2</button>
    </div>
  );
}

# 只在数据更新才执行的useEffect

useEffect(()=>{
    //useEffect在初始化的时候也会执行，但有些操作我并不需要初始化就执行
    //所以想自己封装一个hooks，有依赖项的初始化第一次不执行，后续依赖项更新照常执行
},[state])

就正常当做useEffect使用就行

import { useEffect, useRef } from 'react'

export default function useUpdateEffect(fn, arr) {
  const firstMountRef = useRef(true)
  //初始化不执行fn，后续依赖项更新才执行
  useEffect(() => {
    if (firstMountRef.current) {
      firstMountRef.current = false;
      return;
    } else {
      fn()
    }
  }, arr)
}

# 缓存当前页面/组件的滚动条位置

本身是想实现类似vue的keep-alive组件，但是测试了几种方案效果都不太好，所以干脆想着直接实现两个hooks，一个存储滚动条位置，一个存储状态

import { useEffect, useRef } from "react";
const EXPIRATION_TIME = 24 * 60 * 60 * 1000; // 24 hours

function useScrollPersistence(suffix = "") {
  const scrollContainerRef = useRef(null);
  useEffect(() => {
    const scrollKey = "cacheScroll" + window.location.pathname + "+" + suffix;

    // 检查是否有滚动条
    const hasVerticalScrollbar = suffix
      ? scrollContainerRef.current?.scrollHeight > scrollContainerRef.current?.clientHeight
      : document.documentElement.scrollHeight > window.innerHeight;

    // 恢复滚动位置
    const storedData = localStorage.getItem(scrollKey);
    if (storedData) {
      const { scrollPosition, timestamp } = JSON.parse(storedData);
      //如果没过期且有滚动条
      if (Date.now() < timestamp && hasVerticalScrollbar) {
        if (suffix) {
          scrollContainerRef.current.scrollTop = scrollPosition;
        } else {
          window.scrollTo(0, scrollPosition);
        }
      } else {
        localStorage.removeItem(scrollKey);
      }
    }

    // 存储滚动位置
    const handleScroll = () => {
      const scrollPosition = suffix ? scrollContainerRef.current.scrollTop : window.scrollY;
      const timestamp = Date.now()+EXPIRATION_TIME;
      localStorage.setItem(scrollKey, JSON.stringify({ scrollPosition, timestamp }));
    };

    // 使用防抖函数包装 handleScroll
    const debouncedHandleScroll = debounce(handleScroll, 500);

    if (suffix) {
      scrollContainerRef.current?.addEventListener("scroll", debouncedHandleScroll);
    } else {
      window.addEventListener("scroll", debouncedHandleScroll);
    }

    return () => {
      if (suffix) {
        scrollContainerRef.current?.removeEventListener("scroll", debouncedHandleScroll);
      } else {
        window.removeEventListener("scroll", debouncedHandleScroll);
      }
    };
  }, [suffix]);

  return scrollContainerRef;
}

// 简单的防抖函数
function debounce(func, delay) {
  let timeoutId;
  return function (...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => {
      func.apply(this, args);
    }, delay);
  };
}

export default useScrollPersistence;

路由级页面使用：

import useScrollPersistence from "./useScrollPersistence";

export default function First() {
  useScrollPersistence();  //不传入任何参数，则记录的是window的滚动条，以当前pathname为key
  return (
    <div>
      <div style={{height:'1000px'}}></div>
      <Count />
    </div>
  );
}

组件级使用：

import { useState } from "react";
import useScrollPersistence from "./useScrollPersistence";

export default function Count() {
  const scrollRef = useScrollPersistence('test');    //传入一个id字符串以区分，会以pathname+id为key存储滚动位置
  const [count, setCount] = useState(0);
  return (
    <div ref={scrollRef} style={{height:'200px',background:'red',overflowY:'auto'}}>   //并且把返回的ref传递给组件的需要监听滚动条的层级，一般是最外层
      <div style={{height:'1000px'}}></div>
      <h1>the num is {count}</h1>
      <button
        onClick={() => {
          setCount(count + 1);
        }}
      >
        add
      </button>
    </div>
  );
}

# 缓存当前组件的状态

/* eslint-disable react-hooks/rules-of-hooks */

import { useState, useEffect } from "react";

const cachePath = ["/first"];
const ONE_DAY_IN_MS = 24 * 60 * 60 * 1000; // 一天的毫秒数

function useCacheState(storageKey, name, initState) {
  const prefix = "cacheState" + window.location.pathname;

  // 检查当前路径是否在缓存路径列表中
  const isCachePath = cachePath.includes(window.location.pathname);

  //不是缓存的后面也别执行了
  if (!isCachePath) {
    const [state, setState] = useState(initState);
    return [state, setState];
  }

  // 辅助函数：检查缓存是否过期,返回true则已过期
  const hasCacheExpired = (cache) => {
    const now = Date.now();
    return cache && cache.expiration && now > cache.expiration;
  };

  // 获取并解析缓存值
  const cacheItem = JSON.parse(localStorage.getItem(`${prefix}+${storageKey}`));
  const cacheValue = cacheItem ? cacheItem.value : null;

  // 初始化状态：如果缓存存在且未过期，则使用缓存值，否则使用初始状态
  const [state, setState] = useState(() => {
    if (cacheItem !== null && !hasCacheExpired(cacheItem)) {
      if (cacheValue && cacheValue.hasOwnProperty(name)) {
        return cacheValue[name];
      } else {
        return initState;
      }
    } else {
      if (isCachePath) {
        localStorage.removeItem(`${prefix}+${storageKey}`);
      }
      return initState;
    }
  });


  // 监听状态变化，将状态保存到 localStorage 中，并更新过期时间戳
  useEffect(() => {
    if (isCachePath) {
      const expiration = Date.now() + ONE_DAY_IN_MS;
      if (cacheItem !== null && !hasCacheExpired(cacheItem)) {
        // 更新已有的缓存值
        cacheValue[name] = state;
        localStorage.setItem(
          `${prefix}+${storageKey}`,
          JSON.stringify({ value: cacheValue, expiration })
        );
      } else {
        // 初始化新的缓存值
        const newCacheValue = { [name]: state };
        localStorage.setItem(
          `${prefix}+${storageKey}`,
          JSON.stringify({ value: newCacheValue, expiration })
        );
      }
    }
  }, [state]);

  return [state, setState];
}

export default useCacheState;

组件中使用：需要缓存的值，就使用这个hooks来初始化state，需要传入三个参数

第一个为id字符串，主要是为了防止重复

第二个是存储到localstorage时给的key值

第三个是初始值

import useCacheState from "./useCacheState";

export default function Count(props) {
  const [count, setCount] = useCacheState("countKey", "count", 0);
  const [sum, setSum] = useCacheState("countKey", "sum", 10);
  return (
    <div>
      <h1>the num is {count}</h1>
      <button
        onClick={() => {
          setCount(count + 1);
        }}
      >
        add
      </button>
      <h2>sum is {sum}</h2>
      <button onClick={()=>setSum(sum+2)}>+2</button>
    </div>
  );
}
